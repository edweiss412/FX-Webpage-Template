```sql
-- supabase/migrations/20260520000000_signed_link_admin_rpcs.sql
--
-- M9.5 — Signed-link admin controls (Revoke all + Issue new).
-- Closes M11-E-D1 (HIGH) — see docs/superpowers/plans/2026-04-30-fxav-crew-pages-design/handoffs/M9.5-signed-link-controls.md.
--
-- Two SECURITY DEFINER RPCs gated by public.is_admin() (M9 C9 pattern)
-- and serialized by pg_advisory_xact_lock(hashtext('show:' || drive_file_id))
-- so concurrent admin clicks on the same (show, crew_name) tuple linearize.
--
-- Single-holder advisory-lock rule (AGENTS.md §1.2): the lock is acquired
-- INSIDE the RPC. The Server Action at app/admin/show/[slug]/actions.ts
-- does NOT wrap RPC calls in withShowAdvisoryLock — the meta-test at
-- tests/auth/advisoryLockRpcDeadlock.test.ts pins this topology.
--
-- Apply-twice idempotency: `create or replace function` + `drop function
-- if exists` make re-applying safe.

-- ============================================================================
-- 1. revoke_all_links_rpc(p_show_id uuid, p_crew_name text)
-- ============================================================================
--
-- Sets `revoked_below_version = current_token_version` for the (show_id,
-- crew_name) tuple. Every currently-issued JWT (carrying ≤ current_token_version)
-- becomes ≤ floor and fails the floor check at the redemption path
-- (spec §7.2 step 5).
--
-- Status values:
--   - 'ok' — floor was advanced; returns new floor + row snapshot.
--   - 'no_live_link' — row was already in no-live-link state
--     (current_token_version === revoked_below_version). Idempotent no-op.
--   - 'show_not_found' — show_id didn't resolve to a row.
--   - 'crew_member_not_found' — auth row didn't exist for the tuple.

create or replace function public.revoke_all_links_rpc(
  p_show_id uuid,
  p_crew_name text
)
returns jsonb
language plpgsql
security definer
set search_path = public, pg_temp
as $$
declare
  v_show public.shows%rowtype;
  v_row public.crew_member_auth%rowtype;
begin
  if not public.is_admin() then
    raise exception 'permission denied: signed-link admin RPC requires is_admin()'
      using errcode = '42501';
  end if;

  select * into v_show from public.shows where id = p_show_id;
  if not found then
    return jsonb_build_object('status', 'show_not_found');
  end if;

  -- Single-holder lock on the per-show hashkey.
  perform pg_advisory_xact_lock(hashtext('show:' || v_show.drive_file_id));

  select * into v_row from public.crew_member_auth
   where show_id = p_show_id and crew_name = p_crew_name;
  if not found then
    return jsonb_build_object('status', 'crew_member_not_found');
  end if;

  if v_row.current_token_version = v_row.revoked_below_version then
    return jsonb_build_object('status', 'no_live_link', 'row', row_to_json(v_row));
  end if;

  update public.crew_member_auth
     set revoked_below_version = current_token_version
   where show_id = p_show_id and crew_name = p_crew_name
   returning * into v_row;

  return jsonb_build_object('status', 'ok', 'row', row_to_json(v_row));
end;
$$;

drop function if exists public.revoke_all_links_rpc(uuid, text, uuid);
revoke all on function public.revoke_all_links_rpc(uuid, text) from public;
grant execute on function public.revoke_all_links_rpc(uuid, text)
  to authenticated, service_role;

-- ============================================================================
-- 2. issue_new_link_rpc(p_show_id uuid, p_crew_name text)
-- ============================================================================
--
-- Atomically bumps both current_token_version and max_issued_version to
-- max_issued_version + 1. After the bump, current_token_version is
-- strictly greater than revoked_below_version, so a freshly-minted JWT
-- (carrying the new current_token_version) passes the floor check.
--
-- Status values:
--   - 'ok' — bumped; returns row snapshot including new versions.
--   - 'show_not_found' — show_id didn't resolve.
--   - 'crew_member_not_found' — auth row didn't exist.

create or replace function public.issue_new_link_rpc(
  p_show_id uuid,
  p_crew_name text
)
returns jsonb
language plpgsql
security definer
set search_path = public, pg_temp
as $$
declare
  v_show public.shows%rowtype;
  v_row public.crew_member_auth%rowtype;
begin
  if not public.is_admin() then
    raise exception 'permission denied: signed-link admin RPC requires is_admin()'
      using errcode = '42501';
  end if;

  select * into v_show from public.shows where id = p_show_id;
  if not found then
    return jsonb_build_object('status', 'show_not_found');
  end if;

  perform pg_advisory_xact_lock(hashtext('show:' || v_show.drive_file_id));

  -- Single-statement UPDATE — both columns advance atomically.
  -- Spec §5.2 lines 1095-1097: max_issued_version is a high-water mark;
  -- current_token_version follows it on the bump.
  update public.crew_member_auth
     set current_token_version = max_issued_version + 1,
         max_issued_version = max_issued_version + 1
   where show_id = p_show_id and crew_name = p_crew_name
   returning * into v_row;

  if not found then
    return jsonb_build_object('status', 'crew_member_not_found');
  end if;

  return jsonb_build_object('status', 'ok', 'row', row_to_json(v_row));
end;
$$;

drop function if exists public.issue_new_link_rpc(uuid, text, uuid);
revoke all on function public.issue_new_link_rpc(uuid, text) from public;
grant execute on function public.issue_new_link_rpc(uuid, text)
  to authenticated, service_role;
```

// Append to tests/db/signedLinkAdminRpcs.test.ts

import { execFileSync, spawn } from "node:child_process";
import { randomUUID } from "node:crypto";

const databaseUrl =
  process.env.TEST_DATABASE_URL ?? "postgresql://postgres:postgres@127.0.0.1:54322/postgres";

function runPsql(sql: string): string {
  return execFileSync("psql", [databaseUrl, "-v", "ON_ERROR_STOP=1", "-At"], {
    input: sql,
    encoding: "utf8",
  }).trim();
}

const ADMIN_JWT_SUB = "00000000-0000-0000-0000-000000000020";
function jwtAdmin(email?: string): string {
  const claims: Record<string, unknown> = {
    sub: ADMIN_JWT_SUB,
    app_metadata: { role: "admin" },
  };
  if (email) claims.email = email;
  return JSON.stringify(claims);
}

function seedShowAndCrewSql(driveFileId: string, crewName: string): string {
  return `
    insert into public.shows (title, slug, drive_file_id, published)
    values ('M9.5 test show', ${sqlString(driveFileId)}, ${sqlString(driveFileId)}, true);
    insert into public.crew_member_auth (show_id, crew_name)
    select id, ${sqlString(crewName)} from public.shows
     where drive_file_id = ${sqlString(driveFileId)};
  `;
}

function sqlString(value: string): string {
  return `'${value.replaceAll("'", "''")}'`;
}

describe("revoke_all_links_rpc — behavior (admin JWT, psql, BEGIN...ROLLBACK)", () => {
  test("ok: floor advances to current_token_version on a live row", () => {
    const driveFileId = `m9_5_rt_${randomUUID()}`;
    const crewName = `Alice ${randomUUID()}`;
    const out = runPsql(`
      begin;
      set local role authenticated;
      set local request.jwt.claims = '${jwtAdmin()}';
      ${seedShowAndCrewSql(driveFileId, crewName)}
      update public.crew_member_auth
         set current_token_version = 2, max_issued_version = 2
       where show_id = (select id from public.shows where drive_file_id = ${sqlString(driveFileId)})
         and crew_name = ${sqlString(crewName)};
      select 'result=' || (
        public.revoke_all_links_rpc(
          (select id from public.shows where drive_file_id = ${sqlString(driveFileId)}),
          ${sqlString(crewName)}
        )
      )::text;
      rollback;
    `);
    expect(out).toContain("\"status\" : \"ok\"");
    expect(out).toContain("\"revoked_below_version\" : 2");
    expect(out).toContain("\"current_token_version\" : 2");
  });

  test("no_live_link: idempotent no-op when row is already in no-live-link state", () => {
    const driveFileId = `m9_5_rt_${randomUUID()}`;
    const crewName = `Bob ${randomUUID()}`;
    const out = runPsql(`
      begin;
      set local role authenticated;
      set local request.jwt.claims = '${jwtAdmin()}';
      ${seedShowAndCrewSql(driveFileId, crewName)}
      -- seed default: current_token_version=1, revoked_below_version=0.
      -- First call advances floor to 1 → no-live-link state.
      perform public.revoke_all_links_rpc(
        (select id from public.shows where drive_file_id = ${sqlString(driveFileId)}),
        ${sqlString(crewName)}
      );
      -- Second call: idempotent no-op.
      select 'result=' || (
        public.revoke_all_links_rpc(
          (select id from public.shows where drive_file_id = ${sqlString(driveFileId)}),
          ${sqlString(crewName)}
        )
      )::text;
      rollback;
    `);
    expect(out).toContain("\"status\" : \"no_live_link\"");
  });

  test("show_not_found: missing show_id returns the missing-show sentinel", () => {
    const out = runPsql(`
      begin;
      set local role authenticated;
      set local request.jwt.claims = '${jwtAdmin()}';
      select 'result=' || (
        public.revoke_all_links_rpc(
          '00000000-0000-0000-0000-000000000000'::uuid,
          'Nobody'
        )
      )::text;
      rollback;
    `);
    expect(out).toContain("\"status\" : \"show_not_found\"");
  });

  test("crew_member_not_found: missing crew_member_auth row returns sentinel", () => {
    const driveFileId = `m9_5_rt_${randomUUID()}`;
    const out = runPsql(`
      begin;
      set local role authenticated;
      set local request.jwt.claims = '${jwtAdmin()}';
      insert into public.shows (title, slug, drive_file_id, published)
      values ('M9.5 test show', ${sqlString(driveFileId)}, ${sqlString(driveFileId)}, true);
      -- intentionally omit crew_member_auth INSERT.
      select 'result=' || (
        public.revoke_all_links_rpc(
          (select id from public.shows where drive_file_id = ${sqlString(driveFileId)}),
          'NotASeededName'
        )
      )::text;
      rollback;
    `);
    expect(out).toContain("\"status\" : \"crew_member_not_found\"");
  });

  test("NEGATIVE: non-admin caller is denied at the is_admin() guard", () => {
    const driveFileId = `m9_5_rt_${randomUUID()}`;
    const crewName = `Eve ${randomUUID()}`;

// lib/data/signedLinks.ts
/**
 * lib/data/signedLinks.ts (M9.5)
 *
 * Typed wrappers over the M9.5 signed-link admin RPCs
 * (revoke_all_links_rpc + issue_new_link_rpc). Mirrors the M9 C9
 * adminEmails.ts pattern:
 *   - wrapInfra(name, fn) catches sync+async throws → typed
 *     SignedLinksInfraError per AGENTS.md §1.9.
 *   - Per-RPC translateXResult enforces a status whitelist —
 *     unknown status values surface as infra errors, not silent
 *     success.
 *
 * Why: prevents the M9 C9 R3 class — silent infra fault mistaken
 * for benign action result. The Server Action layer (Phase 3)
 * lets thrown SignedLinksInfraError propagate to the catalog 500
 * surface.
 */
import { createSupabaseServerClient } from "@/lib/supabase/server";

export class SignedLinksInfraError extends Error {
  constructor(message: string) {
    super(message);
    this.name = "SignedLinksInfraError";
  }
}

export type CrewAuthRowSnapshot = {
  current_token_version: number;
  max_issued_version: number;
  revoked_below_version: number;
};

export type IssueLinkOutcome =
  | { kind: "ok"; row: CrewAuthRowSnapshot }
  | { kind: "show_not_found" }
  | { kind: "crew_member_not_found" };

export type RevokeAllLinksOutcome =
  | { kind: "ok"; row: CrewAuthRowSnapshot }
  | { kind: "no_live_link" }
  | { kind: "show_not_found" }
  | { kind: "crew_member_not_found" };

async function wrapInfra<T>(opName: string, fn: () => Promise<T>): Promise<T> {
  try {
    return await fn();
  } catch (err) {
    if (err instanceof SignedLinksInfraError) throw err;
    throw new SignedLinksInfraError(
      `${opName}: ${err instanceof Error ? err.message : String(err)}`,
    );
  }
}

// Append to lib/data/signedLinks.ts

export async function revokeAllLinks(opts: {
  showId: string;
  crewName: string;
}): Promise<RevokeAllLinksOutcome> {
  return wrapInfra("revokeAllLinks", async () => {
    const supabase = await createSupabaseServerClient();
    const { data, error } = await supabase.rpc("revoke_all_links_rpc", {
      p_show_id: opts.showId,
      p_crew_name: opts.crewName,
    });
    if (error) {
      throw new SignedLinksInfraError(`revokeAllLinks.rpc: ${error.message}`);
    }
    return translateRevokeResult(data);
  });
}

function translateRevokeResult(data: unknown): RevokeAllLinksOutcome {
  if (!data || typeof data !== "object" || !("status" in data)) {
    throw new SignedLinksInfraError(
      `revokeAllLinks: malformed RPC payload (missing status): ${JSON.stringify(data)}`,
    );
  }
  const status = (data as { status: unknown }).status;
  if (typeof status !== "string") {
    throw new SignedLinksInfraError(
      `revokeAllLinks: malformed RPC status (not a string): ${JSON.stringify(status)}`,
    );
  }
  switch (status) {
    case "ok": {
      const row = (data as { row: unknown }).row;
      if (!row || typeof row !== "object") {
        throw new SignedLinksInfraError(`revokeAllLinks: ok status missing row payload`);
      }
      return { kind: "ok", row: row as CrewAuthRowSnapshot };
    }
    case "no_live_link":
      return { kind: "no_live_link" };
    case "show_not_found":
      return { kind: "show_not_found" };
    case "crew_member_not_found":
      return { kind: "crew_member_not_found" };
    default:
      throw new SignedLinksInfraError(`revokeAllLinks: unknown status from RPC: ${status}`);
  }
}

"use server";

import { revalidatePath } from "next/cache";
import { requireAdminIdentity } from "@/lib/auth/requireAdmin";
import {
  revokeAllLinks,
  issueNewLink,
} from "@/lib/data/signedLinks";

export type ShowLinkActionResult =
  | { kind: "ok"; code: "ADMIN_LINK_REVOKED_OK" | "ADMIN_LINK_ISSUED_OK" }
  | {
      kind: "refused";
      code:
        | "ADMIN_LINK_NO_LIVE_LINK"
        | "ADMIN_LINK_SHOW_NOT_FOUND"
        | "ADMIN_LINK_CREW_NOT_FOUND";
    };

function readShowAndCrew(formData: FormData): { showId: string; crewName: string } | null {
  const showId = formData.get("showId");
  const crewName = formData.get("crewName");
  if (typeof showId !== "string" || typeof crewName !== "string") return null;
  if (showId.length === 0 || crewName.length === 0) return null;
  return { showId, crewName };
}

export async function revokeAllLinksAction(
  _prev: ShowLinkActionResult | null,
  formData: FormData,
): Promise<ShowLinkActionResult> {
  await requireAdminIdentity();
  const input = readShowAndCrew(formData);
  if (!input) return { kind: "refused", code: "ADMIN_LINK_CREW_NOT_FOUND" };

  const outcome = await revokeAllLinks(input);
  switch (outcome.kind) {
    case "ok":
      revalidatePath("/admin/show/[slug]", "page");
      return { kind: "ok", code: "ADMIN_LINK_REVOKED_OK" };
    case "no_live_link":
      return { kind: "refused", code: "ADMIN_LINK_NO_LIVE_LINK" };
    case "show_not_found":
      return { kind: "refused", code: "ADMIN_LINK_SHOW_NOT_FOUND" };
    case "crew_member_not_found":
      return { kind: "refused", code: "ADMIN_LINK_CREW_NOT_FOUND" };
  }
}

export async function issueNewLinkAction(
  _prev: ShowLinkActionResult | null,
  formData: FormData,

  const input = readShowAndCrew(formData);

}

## Task 3.3: Wire structured audit log in Server Actions (closes handoff AC-9.5-2/3 + Codex R1 MEDIUM)

**Background:** The M9.5 handoff §2 AC-9.5-2 and AC-9.5-3 both state both RPCs "emit an admin-log structured entry." // lib/data/loadShowCrewWithAuth.ts
/**
 * lib/data/loadShowCrewWithAuth.ts (M9.5)
 *
 * Per-show admin page data fetch: load crew_members rows and merge in
 * each member's crew_member_auth version state (current_token_version,
 * max_issued_version, revoked_below_version). Used by
 * app/admin/show/[slug]/page.tsx to render <PerShowCrewSection>.
 *
 * Two separate SELECTs (not a PostgREST embedded join) because the
 * live schema has NO FK from crew_member_auth to crew_members — the
 * correlation is (show_id, crew_name) only.
 *
 * Fail-closed contract (Codex R1 HIGH-1 fix):
 *   - ANY returned-error or thrown-error from EITHER SELECT →
 *     crewLookupFailed=true + crew=[].
 *   - A crew_members row without a matching crew_member_auth row →
 *     authMissing:true with sentinel version values 0/0/0.
 *
 * UI MUST branch on authMissing FIRST — sentinel values 0/0/0 would
 * otherwise satisfy the no-live-link predicate and produce a misleading
 * "Issue first link" label on a row whose auth state is unknown.
 */
import type { SupabaseClient } from "@supabase/supabase-js";

export type CrewRowForLinkPanel = {
  id: string;
  name: string;
  role: string | null;
  authMissing: boolean;
  current_token_version: number;
  max_issued_version: number;
  revoked_below_version: number;
};

export type LoadCrewResult = {
  crew: CrewRowForLinkPanel[];
  crewLookupFailed: boolean;
};

type MemberRow = { id: string; name: string; role: string | null };
type AuthRow = {
  crew_name: string;
  current_token_version: number;
  max_issued_version: number;
  revoked_below_version: number;
};

export async function loadShowCrewWithAuth(
  supabase: SupabaseClient,
  showId: string,
): Promise<LoadCrewResult> {
  let members: MemberRow[] | null;
  let memberErr = false;
  try {
    const { data, error } = await supabase
      .from("crew_members")
      .select("id, name, role")
      .eq("show_id", showId)
      .order("name", { ascending: true });
    if (error !== null) {
      console.error(
        "[loadShowCrewWithAuth] crew_members error:",
        error.message,
      );
      memberErr = true;
      members = null;
    } else {
      members = (data as MemberRow[] | null) ?? null;
    }
  } catch (err) {
    console.error(
      "[loadShowCrewWithAuth] crew_members threw:",
      err instanceof Error ? err.message : String(err),
    );
    memberErr = true;
    members = null;
  }

  let authRows: AuthRow[] | null;
  let authErr = false;
  try {
    const { data, error } = await supabase
      .from("crew_member_auth")
      .select(
        "crew_name, current_token_version, max_issued_version, revoked_below_version",
      )
      .eq("show_id", showId);
    if (error !== null) {
      console.error(
        "[loadShowCrewWithAuth] crew_member_auth error:",
        error.message,
      );
      authErr = true;
      authRows = null;
    } else {
      authRows = (data as AuthRow[] | null) ?? null;
    }
  } catch (err) {
    console.error(
      "[loadShowCrewWithAuth] crew_member_auth threw:",
      err instanceof Error ? err.message : String(err),
    );
    authErr = true;
    authRows = null;
  }

  if (memberErr || authErr) {
    // Fail closed — do NOT render any rows when either lookup failed.
    return { crew: [], crewLookupFailed: true };
  }

  const authByName = new Map(
    (authRows ?? []).map((r) => [r.crew_name, r]),
  );
  const crew: CrewRowForLinkPanel[] = (members ?? []).map((m) => {
    const auth = authByName.get(m.name);
    if (!auth) {
      return {
        id: m.id, name: m.name, role: m.role,
        authMissing: true,
        current_token_version: 0,
        max_issued_version: 0,
        revoked_below_version: 0,
      };
    }
    return {
      id: m.id, name: m.name, role: m.role,
      authMissing: false,
      current_token_version: auth.current_token_version,
      max_issued_version: auth.max_issued_version,
      revoked_below_version: auth.revoked_below_version,
    };
  });
  return { crew, crewLookupFailed: false };
}
